<script>
const message = "another year past, still not dead!!       ðŸª¦ðŸŽ‰\n        \nthe journey begins in a pocket of red    ðŸŸ¥"

function sleep(seconds) {
	return new Promise(resolve => setTimeout(resolve, seconds * 1000))
}

async function revealMessage(event) {
	el = document.getElementById("message")
	for (let character of message) {
		let child
		if (character == "\n") {
			child = document.createElement("br")
		} else {
			child = document.createElement("span")
			child.innerHTML = character === " " ? "&nbsp;" : character
		}
		child.style.animation = "fadein 1s"

		el.appendChild(child)
		await sleep(0.13)
	}
}

revealMessage()
